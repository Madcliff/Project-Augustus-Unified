# Pictish province converts to Scottish
province_event = {
	id = CM.1
	desc = EVTDESC_CM_1
	picture = GFX_evt_stone_church

	trigger = {
		culture = pictish
        NOT = { year = 1000 }
		owner = {
			NOT = {	culture = pictish }
		}
	}
	
	mean_time_to_happen = {
		months = 1200
	}

	option = {
		name = EVTOPTA_CM_1
		culture = scottish
	}
}

# Lombard province converts to Italo-Roman
province_event = {
	id = CM.2
	desc = EVTDESCPA.CM.2
	picture = GFX_evt_stone_church

	trigger = {
		culture = lombard
        NOT = { year = 867 }
		OR = { 
			county = { kingdom = { title = k_lombardy } }
			county = { kingdom = { title = k_sicily } }
        }
        NOT = { county = { title = c_grisons } }
        NOT = { county = { title = c_lombardia } }
	}
	
	mean_time_to_happen = {
		months = 2400
		modifier = {
			factor = 0.5
			owner = {
				culture_group = latin
			}
		}
	}

	option = {
		name = EVTOPTA_CM_2
		culture = old_italian
	}
}

# Old Saxon, Old Frankish or Frisian province converts to Dutch
province_event = {
	id = CM.3
	desc = EVTDESC_CM_3
	picture = GFX_evt_tribal_lands

	trigger = {
		OR = {
			duchy = { title = d_holland }
        		duchy = { title = d_flanders }
        		duchy = { title = d_brabant }
        		duchy = { title = d_gelre }
    		}
    		OR = {
        		culture = old_frankish
        		culture = frisian
        		AND = {
            	    culture = old_saxon
            		has_global_flag = widukind_spawned
        		}
    		}
	}
	
	mean_time_to_happen = {
		months = 1200
		modifier = {
			factor = 0.5
			owner = {
				culture_group = west_germanic
			}
		}
	}

	option = {
		name = EVTOPTA_CM_3
		culture = dutch
	}
}

# Old Saxon province converts to German
character_event = {
	id = CM.4
	desc = EVTDESC_CM_4
	picture = GFX_evt_tribal_lands

	trigger = {
		culture = german
        any_realm_province = {
            culture = old_saxon
        }
	}

	mean_time_to_happen = {
		months = 3600
		modifier = {
			factor = 0.5
			owner = {
				culture_group = central_germanic
			}
		}
	}

	option = {
		name = OPTPA_10278
		random_realm_province = {
            limit = {
			    culture = old_saxon
			}
            culture = german
        }
    }
}

# East Slavic province becomes Russian
province_event = {
	id = CM.11
	desc = EVTDESC_CM_11
	picture = GFX_evt_tribal_lands

	trigger = {
        NOT = { year = 1000 }
		culture_group = east_slavic
		NOT = { culture = russian }
		owner = {
			OR = {
				culture = russian
				AND = {
					NOT = { culture_group = east_slavic }
					NOT = { culture_group = finno_ugric }
				}
				NOT = { religion_group = pagan_group }
				AND = {
					religion_group = pagan_group
					is_reformed_religion = yes
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 1200
	}

	option = {
		name = EVTOPTA_CM_11
		culture = russian
	}
}
