namespace = PA

character_event = {
	id = PA.20001

        hide_window = yes

	trigger = {
		is_feudal = yes
                realm_size = 50
                independent = yes
		higher_tier_than = duke
                NOT = { realm_size = 100 }
                has_law = imperial_administration
		capital_scope = {
			NOT = { has_province_modifier = imperial_rule }
                }
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = OK
                add_character_modifier = { name = imperial_taxes duration = -1 }
		capital_scope = {
		         add_province_modifier = { name = imperial_rule duration = -1 }
                }
        }
}
