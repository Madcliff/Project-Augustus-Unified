# The English Melting Pot
character_event = {
	id = ANI.1
	desc = "EVTDESCANI.55002"
	picture = GFX_evt_stone_church
	
	only_playable = yes
	
	trigger = {
		year = 1100
		culture = english
		primary_title = {
			title = k_ireland
		}
		any_demesne_province = {
			culture = irish
			any_province_holding = {
				NOT = { holding_type = nomad }
			}
		}
	}
	
	mean_time_to_happen = {
		months = 1200
		modifier = {
			factor = 0.75
			learning = 7
		}
		modifier = {
			factor = 0.75
			learning = 9
		}
		modifier = {
			factor = 0.75
			learning = 11
		}
		modifier = {
			factor = 0.75
			learning = 13
		}
		modifier = {
			factor = 1.5
			NOT = { learning = 5 }
		}
		modifier = {
			factor = 1.5
			NOT = { learning = 3 }
		}
		modifier = {
			factor = 2.0
			NOT = { learning = 1 }
		}
	}

	option = {
		name = "ANI.EVTOPTA55002"
		culture = anglo_irish
		random_demesne_province = {
			limit = {
				culture = irish
				any_province_holding = {
					NOT = { holding_type = nomad }
				}
			}
			culture = anglo_irish
		}
	}
}
