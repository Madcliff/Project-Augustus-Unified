# The English Melting Pot
character_event = {
	id = GRB.1
	desc = "EVTDESCGRB.55002"
	picture = GFX_evt_stone_church
	
	only_playable = yes
	
	trigger = {
		culture = bulgarian
		or = {
			primary_title = {
				title = k_greece
			}
			primary_title = {
				title = e_byzantium 
			}
		}
		any_demesne_province = {
			culture = greek
			any_province_holding = {
				NOT = { holding_type = nomad }
			}
		}
	}
	
	mean_time_to_happen = {
		months = 1800
		modifier = {
			factor = 0.75
			learning = 7
		}
		modifier = {
			factor = 0.75
			learning = 9
		}
		modifier = {
			factor = 0.75
			learning = 11
		}
		modifier = {
			factor = 0.75
			learning = 13
		}
		modifier = {
			factor = 1.5
			NOT = { learning = 5 }
		}
		modifier = {
			factor = 1.5
			NOT = { learning = 3 }
		}
		modifier = {
			factor = 2.0
			NOT = { learning = 1 }
		}
	}

	option = {
		name = "GRB.EVTOPTA55002"
		culture = grekiansvo
		random_demesne_province = {
			limit = {
				culture = greek
				any_province_holding = {
					NOT = { holding_type = nomad }
				}
			}
			culture = grekiansvo
		}
	}
}

			}
		}

	}

	option = {
